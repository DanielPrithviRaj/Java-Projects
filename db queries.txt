create table departments (
	id bigint primary key auto_increment,
	department_code int unique not null,
	department_name varchar(50) unique not null,
	created_at timestamp default current_timestamp
);


CREATE TABLE users (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    employee_id VARCHAR(10) UNIQUE NOT NULL,
    fullname VARCHAR(100) NOT NULL,
    password VARCHAR(255) NOT NULL,
    role ENUM('ADMIN','DOCTOR','TECHNICIAN','SECRETARY') NOT NULL,
    department_id BIGINT NULL,
    email VARCHAR(100),
    phone VARCHAR(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (department_id)
        REFERENCES departments(id)
        ON DELETE RESTRICT
);


create table patients (
	id bigint primary key auto_increment,
	patient_hospital_number varchar(20) unique not null,
	full_name varchar(100) not null,
	dob date not null,
	gender ENUM('MALE','FEMALE','OTHER') not null,
	phone Varchar(20) not null,
	blood_group ENUM('A+','A-','B+','B-','AB+','AB-','O+','O-') not null,
	created_at timestamp default current_timestamp
);

create table biopsy_orders(
	id bigint primary key auto_increment,
	
	patient_id bigint not null,
	doctor_id bigint not null,
	department_id bigint not null,

	order_date timestamp default current_timestamp,
	sample_collected_date timestamp null,
	reported_date timestamp null,

	status ENUM('PENDING', 'SAMPLE_COLLECTED', 'REPORTED') DEFAULT 'PENDING',

	notes TEXT,

	Foreign key(patient_id) 
		references patients(id)
		on delete restrict,

	Foreign key(doctor_id)
		references users(id)
		on delete restrict,

	Foreign key(department_id)
		references departments(id)
		on delete restrict
);

CREATE TABLE medical_records (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,

    patient_id BIGINT NOT NULL,
    biopsy_order_id BIGINT UNIQUE NOT NULL,
    doctor_id BIGINT NOT NULL,

    diagnosis VARCHAR(255) NOT NULL,
    report_details TEXT NOT NULL,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (patient_id)
        REFERENCES patients(id)
        ON DELETE RESTRICT,

    FOREIGN KEY (biopsy_order_id)
        REFERENCES biopsy_orders(id)
        ON DELETE RESTRICT,

    FOREIGN KEY (doctor_id)
        REFERENCES users(id)
        ON DELETE RESTRICT
);

CREATE TABLE notifications (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,

    user_id BIGINT NOT NULL,
    biopsy_order_id BIGINT NOT NULL,

    message VARCHAR(255) NOT NULL,
    is_read BOOLEAN DEFAULT FALSE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id)
        REFERENCES users(id)
        ON DELETE CASCADE,

    FOREIGN KEY (biopsy_order_id)
        REFERENCES biopsy_orders(id)
        ON DELETE CASCADE
);
